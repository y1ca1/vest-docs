20/// Two [`Tag`] parsers with the same inner parser but different tag values are disjoint.
21pub broadcast proof fn lemma_disjoint_tag<Inner: SpecParser>(
22    tag1: Tag<Inner, Inner::PVal>,
23    tag2: Tag<Inner, Inner::PVal>,
24)
25    requires
26        tag1.inner == tag2.inner,
27        tag1.tag != tag2.tag,
28    ensures
29        #[trigger] disjoint_domains(tag1, tag2),

33/// Two [`Refined`] parsers with the same inner parser and mutually exclusive predicates are disjoint.

71/// Two tuples are disjoint if their first parsers consume equal bytes and their second parsers are disjoint.

79        forall|input: Seq<u8>| #[trigger]

169/// A productive parser (one that fails on empty input) is disjoint from [`Eof`].
170pub broadcast proof fn lemma_disjoint_eof<P: GoodParser>(p: P, eof: Eof)
171    requires
172        p.spec_parse(seq![]) is None,
173    ensures
174        #[trigger] disjoint_domains(p, eof),
175{
176    assert forall|input: Seq<u8>|
177        #![auto]
178        p.spec_parse(input) is Some && eof.spec_parse(input) is Some implies false by {
179        if eof.spec_parse(input) is Some {
180            assert(input.len() == 0);
181            assert(input == Seq::<u8>::empty());
182        }
183    }
