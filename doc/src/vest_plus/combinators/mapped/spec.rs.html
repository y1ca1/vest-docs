7/// A pair of spec functions forming a bidirectional mapping.
8type IsoFns<In, Out> = (spec_fn(In) -> Out, spec_fn(Out) -> In);
9
10/// A bidirectional mapping between two types (forward for parsing, reverse for
11/// serialization). For roundtrip guarantees, implement and prove [`IsoMapper`].
12pub trait Mapper {
13    /// The input type.
14    type In;
15
16    /// The output type.
17    type Out;
18
19    /// Forward mapping (used during parsing).
20    spec fn spec_map(&self, i: Self::In) -> Self::Out;
21
22    /// Reverse mapping (used during serialization).
23    spec fn spec_map_rev(&self, o: Self::Out) -> Self::In;
24}
25
26impl<In, Out> Mapper for IsoFns<In, Out> {
27    type In = In;
28
29    type Out = Out;
30
31    open spec fn spec_map(&self, i: In) -> Out {
32        (self.0)(i)
33    }
34
35    open spec fn spec_map_rev(&self, o: Out) -> In {
36        (self.1)(o)
37    }
38}
39
40/// A [`Mapper`] proven bijective, thus forming an isomorphism between the input and output types.
41pub trait IsoMapper: Mapper {
42    /// `spec_map_rev(spec_map(i)) == i`.
43    proof fn lemma_map_iso(&self, i: Self::In)
44        ensures
45            self.spec_map_rev(self.spec_map(i)) == i,
46    ;
47
48    /// `spec_map(spec_map_rev(o)) == o`.
49    proof fn lemma_map_iso_rev(&self, o: Self::Out)
50        ensures
51            self.spec_map(self.spec_map_rev(o)) == o,
52    ;
53}

55impl<Inner, M> SpecParser for super::Mapped<Inner, M> where
56    Inner: SpecParser,
57    M: Mapper<In = Inner::PVal>,
58 {
59    type PVal = M::Out;
60
61    open spec fn spec_parse(&self, ibuf: Seq<u8>) -> Option<(int, M::Out)> {
62        match self.inner.spec_parse(ibuf) {
63            Some((n, v)) => Some((n, self.mapper.spec_map(v))),
64            None => None,
65        }
66    }
67}

69impl<Inner, M> SoundParser for super::Mapped<Inner, M> where
70    Inner: SoundParser,
71    M: IsoMapper<In = Inner::PVal>,
72 {
73    open spec fn sound_inv(&self) -> bool {
74        self.inner.sound_inv()
75    }
76
77    proof fn lemma_parse_safe(&self, ibuf: Seq<u8>) {
78        self.inner.lemma_parse_safe(ibuf);
79    }
80
81    proof fn lemma_parse_sound_consumption(&self, ibuf: Seq<u8>) {
82        self.inner.lemma_parse_sound_consumption(ibuf);
83        if let Some((n, inner_v)) = self.inner.spec_parse(ibuf) {
84            self.mapper.lemma_map_iso(inner_v);
85        }
86    }
87
88    proof fn lemma_parse_sound_value(&self, ibuf: Seq<u8>) {
89        self.inner.lemma_parse_sound_value(ibuf);
90        if let Some((n, inner_v)) = self.inner.spec_parse(ibuf) {
91            self.mapper.lemma_map_iso(inner_v);
92        }
93    }
94}
