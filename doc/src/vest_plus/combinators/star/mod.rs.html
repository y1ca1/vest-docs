12/// Kleene star combinator: greedy zero-or-more repetition, consuming/producing `Seq<A::PVal>`.
13///
14/// Parsing semantics: always succeeds (may return an empty sequence). Stops when `A` fails or
15/// consumes zero bytes.
16///
17/// ## Consistency
18///
19/// A sequence `s` is consistent with `Star<A>` iff every element of `s` is consistent with `A`.
20///
21/// ## Note
22///
23/// This combinator is mostly used *internally* to specify [`Repeat<A, B>`], which is
24/// able to disambiguate `A` and `B` and hence more compositional.
25pub struct Star<A> {
26    /// The inner combinator to repeat.
27    pub inner: A,
28}
29
30/// Zero-or-more `A` followed by terminator `B`: sugar for `(Star<A>, B)`.
31///
32/// ## Unambiguity
33///
34/// Requires `disjoint_domains(A, B)`.
35pub struct Repeat<A, B>(pub A, pub B);
36
37/// Exactly `N` repetitions of combinator `C` (`N` is a runtime value).
38pub struct RepeatN<C, N = u8>(pub N, pub C);
39
40/// Exactly `N` repetitions of combinator `C` (`N` is a compile-time constant).
41pub struct Array<const N: usize, C>(pub C);
42
43} // verus!
