6/// Defines one level of a recursive format for use with [`super::Fix`].
7pub trait RecBody {
8    /// The type of values parsed/serialized by this recursive format.
9    type Val;
10
11    /// Parser for one level, given a recursive callback.
12    spec fn parse_body(&self, rec: ParserFnSpec<Self::Val>) -> ParserFnSpec<Self::Val>;
13
14    /// Byte-length function for one level, given a recursive callback.
15    spec fn byte_len_body(&self, rec: ByteLenFnSpec<Self::Val>) -> ByteLenFnSpec<Self::Val>;
16
17    /// Consistency predicate for one level, given a recursive callback.
18    spec fn consistent_body(&self, rec: PredFnSpec<Self::Val>) -> PredFnSpec<Self::Val>;
19
20    /// [`good_parser`] is preserved through one level of unfolding.
21    proof fn lemma_body_preservation(
22        &self,
23        p_rec: ParserFnSpec<Self::Val>,
24        c_rec: PredFnSpec<Self::Val>,
25        b_rec: ByteLenFnSpec<Self::Val>,
26    )
27        requires
28            good_parser(p_rec, c_rec, b_rec),
29        ensures
30            good_parser(
31                self.parse_body(p_rec),
32                self.consistent_body(c_rec),
33                self.byte_len_body(b_rec),
34            ),
35    ;
36}

105/// The functional version of [`GoodParser`].
106pub open spec fn good_parser<T>(
107    parser: ParserFnSpec<T>,
108    consistent: PredFnSpec<T>,
109    byte_len: ByteLenFnSpec<T>,
110) -> bool {
111    forall|input: Seq<u8>| #[trigger]
112        parser(input) matches Some((n, v)) ==> {
113            &&& 0 <= n <= input.len()
114            &&& consistent(v)
115            &&& byte_len(v) == n
116        }
117}

119impl<const LIMIT: usize, Body: RecBody> super::Fix<LIMIT, Body> {
120    /// Parse with a given amount of gas. Unfolds one level via
121    /// `body.parse_body` with a callback that recurses at `gas - 1`.
122    pub open spec fn spec_parse_with_gas(&self, gas: nat, input: Seq<u8>) -> Option<
123        (int, Body::Val),
124    >
125        decreases gas, 1nat,
126    {
127        self.0.parse_body(self.parse_callback(gas))(input)
128    }
129
130    /// Consistency check with given gas.
131    pub open spec fn consistent_with_gas(&self, gas: nat, v: Body::Val) -> bool
132        decreases gas, 1nat,
133    {
134        self.0.consistent_body(self.consistent_callback(gas))(v)
135    }
136
137    /// Byte-length computation with given gas.
138    pub open spec fn byte_len_with_gas(&self, gas: nat, v: Body::Val) -> nat
139        decreases gas, 1nat,
140    {
141        self.0.byte_len_body(self.byte_len_callback(gas))(v)
142    }
143
144    /// Recursive parser callback: recurses at `gas - 1`, returns `None` at zero.
145    pub open spec fn parse_callback(&self, gas: nat) -> ParserFnSpec<Body::Val>
146        decreases gas, 0nat,
147    {
148        |ibuf: Seq<u8>|
149            {
150                if gas > 0 {
151                    self.spec_parse_with_gas((gas - 1) as nat, ibuf)
152                } else {
153                    None
154                }
155            }
156    }
157
158    /// Recursive consistency callback: recurses at `gas - 1`, returns `false` at zero.
159    pub open spec fn consistent_callback(&self, gas: nat) -> PredFnSpec<Body::Val>
160        decreases gas, 0nat,
161    {
162        |vv: Body::Val|
163            {
164                if gas > 0 {
165                    self.consistent_with_gas((gas - 1) as nat, vv)
166                } else {
167                    false
168                }
169            }
170    }
171
172    /// Recursive byte-length callback: recurses at `gas - 1`, returns `0` at zero.
173    pub open spec fn byte_len_callback(&self, gas: nat) -> ByteLenFnSpec<Body::Val>
174        decreases gas, 0nat,
175    {
176        |vv: Body::Val|
177            {
178                if gas > 0 {
179                    self.byte_len_with_gas((gas - 1) as nat, vv)
180                } else {
181                    0
182                }
183            }
184    }
185
186    /// Inductive proof that `spec_parse_with_gas` satisfies [`good_parser`].
187    proof fn good_parser_by_induction(&self, gas: nat, input: Seq<u8>, n: int, v: Body::Val)
188        ensures
189            self.spec_parse_with_gas(gas, input) == Some((n, v)) ==> {
190                &&& 0 <= n <= input.len()
191                &&& self.consistent_with_gas(gas, v)
192                &&& self.byte_len_with_gas(gas, v) == n
193            },
194        decreases gas,
195    {
196        // vacuous case
197        if !(self.spec_parse_with_gas(gas, input) == Some((n, v))) {
198            return ;
199        }
200        let callback_p = self.parse_callback(gas);
201        let callback_c = self.consistent_callback(gas);
202        let callback_b = self.byte_len_callback(gas);
203
204        // establish good_parser(callback_p, callback_c, callback_b)
205        assert forall|rem: Seq<u8>| #[trigger]
206            callback_p(rem) matches Some((nn, vv)) ==> {
207                &&& 0 <= nn <= rem.len()
208                &&& callback_c(vv)
209                &&& callback_b(vv) == nn
210            } by {
211            if let Some((nn, vv)) = callback_p(rem) {
212                self.good_parser_by_induction((gas - 1) as nat, rem, nn, vv);
213                // IH gives: 0 <= nn <= rem.len(), consistent_with_gas(gas - 1, vv), byte_len_with_gas(gas - 1, vv) == nn
214                assert(0 <= nn <= rem.len());
215
216                assert(self.spec_parse_with_gas((gas - 1) as nat, rem) == Some((nn, vv)));
217                assert(self.consistent_with_gas((gas - 1) as nat, vv) == callback_c(vv));
218                assert(self.byte_len_with_gas((gas - 1) as nat, vv) == callback_b(vv));
219                assert(callback_c(vv));
220                assert(callback_b(vv) == nn);
221            }
222        }
223
224        assert(good_parser(callback_p, callback_c, callback_b));
225        self.0.lemma_body_preservation(callback_p, callback_c, callback_b);
226        // By definition
227        assert(self.spec_parse_with_gas(gas, input) == self.0.parse_body(callback_p)(input));
228        assert(self.consistent_with_gas(gas, v) == self.0.consistent_body(callback_c)(v));
229        assert(self.byte_len_with_gas(gas, v) == self.0.byte_len_body(callback_b)(v));
230    }
231}

233impl<const LIMIT: usize, Body: RecBody> GoodParser for super::Fix<LIMIT, Body> {
234    proof fn lemma_parse_len_bound(&self, ibuf: Seq<u8>) {
235        if let Some((n, v)) = self.spec_parse(ibuf) {
236            self.good_parser_by_induction(LIMIT as nat, ibuf, n, v);
237        }
238    }
239
240    proof fn lemma_parse_byte_len(&self, ibuf: Seq<u8>) {
241        if let Some((n, v)) = self.spec_parse(ibuf) {
242            self.good_parser_by_induction(LIMIT as nat, ibuf, n, v);
243        }
244    }
245
246    proof fn lemma_parse_consistent(&self, ibuf: Seq<u8>) {
247        if let Some((n, v)) = self.spec_parse(ibuf) {
248            self.good_parser_by_induction(LIMIT as nat, ibuf, n, v);
249        }
250    }

260/// Example recursive value type: nested braces `{...}` or empty `\0`.
261pub enum NestedBracesT {
262    /// A brace-wrapped recursive value: `'{' inner '}'`.
263    Brace(Box<NestedBracesT>),
264    /// The empty (base case) value: `'\0'`.
265    Eps,
266}
267
268/// Mapper between `Sum<NestedBracesT, ()>` and `NestedBracesT`.
269pub struct NestedBracesMapper;
270
271impl Mapper for NestedBracesMapper {
272    type In = Sum<NestedBracesT, ()>;
273
274    type Out = NestedBracesT;
275
276    open spec fn spec_map(&self, i: Self::In) -> Self::Out {
277        match i {
278            Sum::Inl(inner) => NestedBracesT::Brace(Box::new(inner)),
279            Sum::Inr(()) => NestedBracesT::Eps,
280        }
281    }
282
283    open spec fn spec_map_rev(&self, o: Self::Out) -> Self::In {
284        match o {
285            NestedBracesT::Brace(inner) => Sum::Inl(*inner),
286            NestedBracesT::Eps => Sum::Inr(()),
287        }
288    }
289}
290
291impl IsoMapper for NestedBracesMapper {
292    proof fn lemma_map_iso(&self, i: Self::In) {
293        match i {
294            Sum::Inl(_) => {},
295            Sum::Inr(u) => {
296                assert(u == ());
297            },
298        }
299    }
300
301    proof fn lemma_map_iso_rev(&self, o: Self::Out) {
302    }
303}

305/// One level of the nested-braces format: `'{' rec '}' | '\0'`.
306pub open spec fn nested_braces_body<Rec>(rec: Rec) -> Mapped<
307    Choice<Terminated<Preceded<Tag<U8, u8>, Rec>, Tag<U8, u8>>, Tag<U8, u8>>,
308    NestedBracesMapper,
309> {
310    Mapped {
311        inner: Choice(
312            Terminated(
313                Preceded(Tag { inner: U8, tag: 0x7Bu8 }, rec),
314                Tag { inner: U8, tag: 0x7Du8 },
315            ),
316            Tag { inner: U8, tag: 0x00u8 },
317        ),
318        mapper: NestedBracesMapper,
319    }
320}

322/// [`RecBody`] for the nested-braces example.
323pub struct NestedBracesBody;
324
325impl RecBody for NestedBracesBody {
326    type Val = NestedBracesT;
327
328    open spec fn parse_body(&self, rec: ParserFnSpec<Self::Val>) -> ParserFnSpec<Self::Val> {
329        |input: Seq<u8>| nested_braces_body(rec).spec_parse(input)
330    }
331
332    open spec fn byte_len_body(&self, rec: ByteLenFnSpec<Self::Val>) -> ByteLenFnSpec<Self::Val> {
333        |v: Self::Val| nested_braces_body(rec).byte_len(v)
334    }
335
336    open spec fn consistent_body(&self, rec: PredFnSpec<Self::Val>) -> PredFnSpec<Self::Val> {
337        |v: Self::Val|
338            nested_braces_body(rec).consistent(v)
339        // |v: Self::Val|
340        //     exists|p: ParserFnSpec<Self::Val>, b: ByteLenFnSpec<Self::Val>| #[trigger]
341        //         nested_braces_body((p, rec, b)).consistent(v)
342
343    }
344
345    proof fn lemma_body_preservation(
346        &self,
347        p_rec: ParserFnSpec<Self::Val>,
348        c_rec: PredFnSpec<Self::Val>,
349        b_rec: ByteLenFnSpec<Self::Val>,
350    ) {
351        assert forall|input: Seq<u8>| #[trigger]
352            self.parse_body(p_rec)(input) matches Some((n, v)) ==> {
353                &&& 0 <= n <= input.len()
354                &&& self.consistent_body(c_rec)(v)
355                &&& self.byte_len_body(b_rec)(v) == n
356            } by {
357            let body = nested_braces_body((p_rec, c_rec, b_rec));
358            body.lemma_parse_len_bound(input);
359            body.lemma_parse_byte_len(input);
360            body.lemma_parse_consistent(input);
361            let body_cons = nested_braces_body(c_rec);
362            assert(body_cons.inner.0.0.0.consistent(()));
363            assert(body_cons.inner.0.1.consistent(()));
364        }
365    }
366}

368proof fn nested_braces_good_parser() {
369    let nested_braces = super::Fix::<10, _>(NestedBracesBody);
370
371    let input = seq![0x7Bu8, 0x00u8, 0x7Du8];
372
373    assert(nested_braces.spec_parse(input) == Some(
374        (3int, NestedBracesT::Brace(Box::new(NestedBracesT::Eps))),
375    )) by {
376        let body10 = nested_braces_body(nested_braces.parse_callback(10nat));
377        assert(body10.spec_parse(input) == Some(
378            (3int, NestedBracesT::Brace(Box::new(NestedBracesT::Eps))),
379        ));
380    };
381
382    nested_braces.lemma_parse_byte_len(input);
383    nested_braces.lemma_parse_consistent(input);
384    nested_braces.lemma_parse_len_bound(input);
385}
