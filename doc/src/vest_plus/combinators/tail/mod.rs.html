12/// Tail combinator: denotes the "tail" of the format, useful for under-specification.
13///
14/// Parsing semantics: consumes and return all remaining bytes (even if the input is empty).
15///
16/// ## Note
17///
18/// The DPS serialization replaces (not prepends to) the output buffer,
19/// so `Tail` should only appear at the end of a format (and the trait system enforces this).
20pub struct Tail;
21
22/// End-of-file combinator: denotes the "EOF".
23///
24/// Parsing semantics: succeeds only if the input is empty, producing `()`.
25///
26/// Implements [`AdmitsUniqueVal`](crate::core::spec::AdmitsUniqueVal).
27///
28/// ## Note
29///
30/// The DPS serialization always replaces the output buffer with the empty sequence, so `Eof`
31/// should only appear at the end of a format (and the trait system enforces this).
32pub struct Eof;
33
34/// Sugar for `Optional(C, Eof)`.
35pub struct OptionalEof<C>(pub C);
36
37/// Sugar for `Repeat(C, Eof)`.
38pub struct RepeatUtilEof<C>(pub C);

40} // verus!
