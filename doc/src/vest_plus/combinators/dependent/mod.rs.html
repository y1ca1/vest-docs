12/// A family of dependent combinators indexed by a key type.
13///
14/// Used to constrain the `Tail` combinator in a [`Bind<Head, Tail>`].
15///
16/// While possible, most users will use the provided dependent combinators ([`TVOr`], [`TLVal`],
17/// [`VariedLen`], etc.) rather than implementing this trait directly.
18pub trait DepCombinator {
19    /// The type of keys parsed by the head combinator/to be recovered during serialization.
20    type Key;
21
22    /// The type of values consumed/produced by the body combinator.
23    type Val;
24
25    /// The type of the body combinator produced by `apply`.
26    type Body: Consistency<Val = Self::Val>;
27
28    /// Given a key, produce the body combinator for that key.
29    spec fn apply(&self, key: Self::Key) -> Self::Body;
30
31    /// Given a body value, recover the key used to produce the body combinator.
32    spec fn recover(&self, value: Self::Val) -> Self::Key;
33
34    /// Recover must agree with any key that makes the body's value consistent.
35    proof fn lemma_recover_consistent(&self, key: Self::Key, value: Self::Val)
36        ensures
37            self.apply(key).consistent(value) ==> self.recover(value) == key,
38    ;
39}

41/// Dependent sequential combinator with deterministic key recovery.
42///
43/// Parsing semantics: parse `Head` to get a key, then parse the
44/// body via `Tail::apply(key)`. Only the body value is returned.
45///
46/// The key is recovered via `Tail::recover(value)` during serialization.
47///
48/// ## Consistency
49///
50/// A value `v: Tail::Val` is consistent with `Bind(Head, Tail)` iff
51/// ```rust
52/// let key = self.1.recover(v);
53/// self.0.consistent(key) && self.1.apply(key).consistent(v)
54/// ```
55///
56/// ## Unambiguity
57///
58/// ```rust
59/// self.0.unambiguous() &&
60/// forall|key: Head::PVal| #[trigger] (self.1.apply(key)).unambiguous()
61/// ```
62pub struct Bind<Head, Tail>(pub Head, pub Tail);
