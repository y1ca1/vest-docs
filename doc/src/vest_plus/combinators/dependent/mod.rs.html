13/// A family of dependent combinators indexed by a key type.
14///
15/// Used to constrain the `Tail` combinator in a [`Bind<Head, Tail>`].
16///
17/// While possible, most users will use the provided dependent combinators ([`TVOr`], [`TLVal`],
18/// [`VariedLen`], etc.) rather than implementing this trait directly.
19pub trait DepCombinator {
20    /// The type of keys parsed by the head combinator/to be recovered during serialization.
21    type Key;
22
23    /// The type of values consumed/produced by the body combinator.
24    type Val;
25
26    /// The type of the body combinator produced by `apply`.
27    type Body: Consistency<Val = Self::Val>;
28
29    /// Given a key, produce the body combinator for that key.
30    spec fn apply(&self, key: Self::Key) -> Self::Body;
31
32    /// Given a body value, recover the key used to produce the body combinator.
33    spec fn recover(&self, value: Self::Val) -> Self::Key;
34
35    /// Recover must agree with any key that makes the body's value consistent.
36    proof fn lemma_recover_consistent(&self, key: Self::Key, value: Self::Val)
37        ensures
38            self.apply(key).consistent(value) ==> self.recover(value) == key,
39    ;
40}

42/// Dependent sequential combinator with deterministic key recovery.
43///
44/// Parsing semantics: parse `Head` to get a key, then parse the
45/// body via `Tail::apply(key)`. Only the body value is returned.
46///
47/// The key is recovered via `Tail::recover(value)` during serialization.
48///
49/// ## Consistency
50///
51/// A value `v: Tail::Val` is consistent with `Bind(Head, Tail)` iff
52/// ```rust
53/// let key = self.1.recover(v);
54/// self.0.consistent(key) && self.1.apply(key).consistent(v)
55/// ```
56///
57/// ## Unambiguity
58///
59/// ```rust
60/// self.0.unambiguous() &&
61/// forall|key: Head::PVal| #[trigger] (self.1.apply(key)).unambiguous()
62/// ```
63pub struct Bind<Head, Tail>(pub Head, pub Tail);
