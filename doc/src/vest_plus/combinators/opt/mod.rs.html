12/// Optional combinator: denotes an optional field.
13///
14/// Parsing semantics: tries `A`, returning `Some(a)` on success; on failure, returns `None` without consuming input.
15///
16/// Serialization semantics: if the value is `Some(a)`, serializes `a` with `A`; if the value is `None`, produces no output.
17///
18/// ## Consistency
19///
20/// A value `v` is consistent with `Opt<A>` iff either `v` is consistent with `A` or `v` is `None`.
21///
22/// ## Note
23///
24/// This combinator is mostly used *internally* to specify [`Optional<A, B>`], which is
25/// able to disambiguate `A` and `B` and hence more compositional.
26pub struct Opt<A>(pub A);
27
28/// Optional field with an arbitrary continuation, defined as `(Opt<A>, B)`.
29///
30/// ## Unambiguity
31///
32/// Requires `disjoint_domains(A, B)`.
33pub struct Optional<A, B>(pub A, pub B);
34
35} // verus!
