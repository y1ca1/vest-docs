6impl<A: SPRoundTripDps + GoodSerializerDps, B: SPRoundTripDps> SPRoundTripDps for (A, B) {
7    proof fn theorem_serialize_dps_parse_roundtrip(&self, v: Self::T, obuf: Seq<u8>) {
8        let serialized1 = self.1.spec_serialize_dps(v.1, obuf);
9        let serialized0 = self.0.spec_serialize_dps(v.0, serialized1);
10        self.1.theorem_serialize_dps_parse_roundtrip(v.1, obuf);
11        self.0.theorem_serialize_dps_parse_roundtrip(v.0, serialized1);
12        self.0.lemma_serialize_dps_buf(v.0, serialized1);
13        self.0.lemma_serialize_dps_len(v.0, serialized1);
14        if let Some((n0, v0)) = self.0.spec_parse(serialized0) {
15            assert(n0 == serialized0.len() - serialized1.len());
16            assert(serialized0.skip(n0) == serialized1);
17        }
18    }

26impl<A: NonMalleable, B: NonMalleable> NonMalleable for (A, B) {
27    proof fn lemma_parse_non_malleable(&self, buf1: Seq<u8>, buf2: Seq<u8>) {
28        if let Some((n1, v1)) = self.spec_parse(buf1) {
29            if let Some((n2, v2)) = self.spec_parse(buf2) {
30                if v1 == v2 {
31                    let (n1a, _) = self.0.spec_parse(buf1)->0;
32                    let (n2a, _) = self.0.spec_parse(buf2)->0;
33                    let (n1b, _) = self.1.spec_parse(buf1.skip(n1a))->0;
34                    let (n2b, _) = self.1.spec_parse(buf2.skip(n2a))->0;
35                    self.0.lemma_parse_len_bound(buf1);
36                    self.0.lemma_parse_len_bound(buf2);
37                    self.1.lemma_parse_len_bound(buf1.skip(n1a));
38                    self.1.lemma_parse_len_bound(buf2.skip(n2a));
39                    self.0.lemma_parse_non_malleable(buf1, buf2);
40                    self.1.lemma_parse_non_malleable(buf1.skip(n1a), buf2.skip(n2a));
41                    assert(n1 == n1a + n1b && n2 == n2a + n2b);
42                    assert(buf1.take(n1) == buf2.take(n2)) by {
43                        assert(buf1.take(n1) == buf1.take(n1a) + buf1.skip(n1a).take(n1b));
44                        assert(buf2.take(n2) == buf2.take(n2a) + buf2.skip(n2a).take(n2b));
45                    }
46                }
47            }
48        }
49    }

62impl<A: NoLookAhead, B: NoLookAhead> NoLookAhead for (A, B) {
63    proof fn lemma_no_lookahead(&self, i1: Seq<u8>, i2: Seq<u8>) {
64        broadcast use vstd::seq_lib::group_seq_properties;
65
66        if let Some((n, v)) = self.spec_parse(i1) {
67            if 0 <= n <= i2.len() {
68                if i2.take(n) == i1.take(n) {
69                    if let Some((n1, v1)) = self.0.spec_parse(i1) {
70                        if let Some((n2, v2)) = self.1.spec_parse(i1.skip(n1)) {
71                            self.lemma_parse_len_bound(i1);
72                            self.0.lemma_parse_len_bound(i1);
73                            self.1.lemma_parse_len_bound(i1.skip(n1));
74                            assert(i2.take(n1) == i1.take(n1));
75                            self.0.lemma_no_lookahead(i1, i2);
76                            assert(i2.skip(n1).take(n2) == i1.skip(n1).take(n2)) by {
77                                lemma_take_skip(i1, n1, n2);
78                                lemma_take_skip(i2, n1, n2);
79                            };
80                            self.1.lemma_no_lookahead(i1.skip(n1), i2.skip(n1));
81                        }
82                    }
83                }
84            }
85        }
86    }

89impl<A, B> EquivSerializersGeneral for (A, B) where
90    A: EquivSerializersGeneral,
91    B: EquivSerializersGeneral,
92 {
93    proof fn lemma_serialize_equiv(&self, v: Self::SVal, obuf: Seq<u8>) {
94        let obuf1 = self.1.spec_serialize_dps(v.1, obuf);
95
96        self.1.lemma_serialize_equiv(v.1, obuf);
97        self.0.lemma_serialize_equiv(v.0, obuf1);
98
99        // From self.1.lemma_serialize_equiv:
100        // self.1.spec_serialize_dps(v.1, obuf) == self.1.spec_serialize(v.1) + obuf
101        // So: obuf1 == self.1.spec_serialize(v.1) + obuf
102
103        // From self.0.lemma_serialize_equiv:
104        // self.0.spec_serialize_dps(v.0, obuf1) == self.0.spec_serialize(v.0) + obuf1
105
106        // Therefore:
107        // spec_serialize_dps(v, obuf) = self.0.spec_serialize_dps(v.0, obuf1)
108        //                              = self.0.spec_serialize(v.0) + obuf1
109        //                              = self.0.spec_serialize(v.0) + self.1.spec_serialize(v.1) + obuf
110        //                              = spec_serialize(v) + obuf
111    }
112}
